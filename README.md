## Install

```bash
npm install --save react-tabletrim
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-tabletrim'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
